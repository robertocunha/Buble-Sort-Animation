    function sort(d) {
        for(var i = 0; i < d.length; i++) {
            for(var j = i + 1; j < d.length; j++) {
                if(d[i] > d[j]) {
                    const temp = d[i]
                    d[i] = d[j]
                    d[j] = temp
                }
            }
        }
    }
