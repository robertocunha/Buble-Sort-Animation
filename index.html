    function sort(bars) {
        for(let i = 0; i < bars.length; i++) {
            for(let j = i + 1; j < bars.length; j++) {
                if(bars[i] > bars[j]) {
                    const temp = bars[i]
                    bars[i] = bars[j]
                    bars[j] = temp
                }
            }
        }
    }
